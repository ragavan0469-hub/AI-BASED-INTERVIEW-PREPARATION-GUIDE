# --- Pydantic Models for Structured LLM Output ---
class QuizQuestion(BaseModel):
    """A single multiple-choice question with options and the correct answer."""
    question: str = Field(description="The generated multiple-choice question.")
    options: List[str] = Field(description="Exactly 4 plausible but distinct options for the question.")
    correct_answer: str = Field(description="The correct option *text* (must be one of the options).")

# NEW Pydantic Model to structure project extraction
class ProjectContext(BaseModel):
    """A list of project titles and their corresponding detailed descriptions."""
    projects: List[Dict[str, str]] = Field(description="A list where each item is a dictionary with 'title' and 'description'.")

def highlight_projects_in_pdf(pdf_path, output_path, project_labels):
    doc = fitz.open(pdf_path)
    colors = {'FIT': (0, 0.7, 0), 'CHANGE': (0.9, 0, 0)}
    for page in doc:
        page_text = page.get_text()
        for proj_name, data in project_labels.items():
            label = data.get("label", "CHANGE")
            lower_text = page_text.lower()
            lower_proj = proj_name.lower()
            start = 0
            while True:
                idx = lower_text.find(lower_proj, start)
                if idx == -1:
                    break
                text_instances = page.search_for(page_text[idx:idx+len(proj_name)])
                for inst in text_instances:
                    annot = page.add_highlight_annot(inst)
                    annot.set_colors(stroke=colors.get(label, (0.9, 0, 0)))
                    annot.update()
                start = idx + len(proj_name)
    Path(output_path).parent.mkdir(parents=True, exist_ok=True)
    doc.save(output_path)
    doc.close()

def save_to_dashboard(resume_name, role, job_desc, project_labels):
    summary = {
        "resume_name": resume_name,
        "role": role,
        "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
        "fit_count": sum(1 for v in project_labels.values() if v.get("label") == "FIT"),
        "change_count": sum(1 for v in project_labels.values() if v.get("label") == "CHANGE"),
        "projects": json.dumps(project_labels, indent=2),
        "job_description": job_desc[:200] + ("..." if len(job_desc) > 200 else "")
    }
    df = pd.DataFrame([summary])
    if os.path.exists(DATA_PATH):
        df.to_csv(DATA_PATH, mode="a", header=False, index=False)
    else:
        df.to_csv(DATA_PATH, index=False)

def load_dashboard_data():
    if not os.path.exists(DATA_PATH):
        return pd.DataFrame(columns=["resume_name", "role", "timestamp", "fit_count", "change_count", "job_description"])
    return pd.read_csv(DATA_PATH)

def safe_json_parse(text, default):
    try:
        cleaned = text.replace("```json", "").replace("```", "").strip()
        return json.loads(cleaned)
    except Exception:
        return default


# --- MODIFIED: resume_analysis_pipeline to run NEW context extractor ---
def resume_analysis_pipeline(temp_pdf_path, job_description, role):
    if not llm_text:
        return "LLM not configured.", "/tmp/dummy.pdf", {}, [], "", []

    highlighted_pdf_path = "/tmp/resume_highlighted.pdf"
    with open(temp_pdf_path, "rb") as f:
        reader = PyPDF2.PdfReader(f)
        resume_text = "".join((page.extract_text() or "") + "\n" for page in reader.pages)

    # 1. Project Title Extraction (unchanged)
    extract_task = Task(
        description=f"Extract project titles from this resume text:\n{resume_text[:6000]}",
        agent=extract_projects_agent,
        expected_output="Return a JSON array of project titles, e.g. ['Project A', 'Project B', 'Project C']"
    )
    crew_extract = Crew(agents=[extract_projects_agent], tasks=[extract_task], process=Process.sequential)
    raw_extract = str(crew_extract.kickoff())
    project_titles = safe_json_parse(raw_extract, [])

    # 2. NEW: Project Context Extraction for question generation
    context_task = Task(
        description=f"Extract titles and full descriptions for projects matching this list: {project_titles}. Use the full resume text:\n{resume_text[:6000]}",
        agent=context_extractor_agent,
        expected_output="A valid JSON object matching the ProjectContext schema.",
        output_pydantic=ProjectContext
    )
    crew_context = Crew(agents=[context_extractor_agent], tasks=[context_task], process=Process.sequential)
    try:
        context_result = crew_context.kickoff()
        # Convert Pydantic object to a simple list of dicts for state storage
        project_context_list = context_result.projects
    except Exception as e:
        print(f"Error extracting project context: {e}")
        project_context_list = []

    # 3. Project Evaluation (unchanged)
    evaluate_task = Task(
        description=f"Evaluate these projects {project_titles} for the role '{role}' and JD: {job_description}",
        agent=evaluate_projects_agent,
        expected_output="Return a JSON dictionary where each key is a project title and each value is {'label': 'FIT' or 'CHANGE', 'reason': 'short justification'}"
    )
    crew_eval = Crew(agents=[evaluate_projects_agent], tasks=[evaluate_task], process=Process.sequential)
    eval_raw = str(crew_eval.kickoff())
    project_labels = safe_json_parse(eval_raw, {})

    # 4. Resume Improvement (unchanged)
    improve_task = Task(
        description=f"Suggest improvements for role '{role}' based on {json.dumps(project_labels)}",
        agent=improvements_agent,
        expected_output="Return concise bullet-point suggestions for improving resume projects"
    )
    crew_improve = Crew(agents=[improvements_agent], tasks=[improve_task], process=Process.sequential)
    improvements = str(crew_improve.kickoff())

    highlight_projects_in_pdf(temp_pdf_path, highlighted_pdf_path, project_labels)
    save_to_dashboard(os.path.basename(temp_pdf_path), role, job_description, project_labels)

    # MODIFIED: Return project_context_list instead of raw resume_text
    return improvements, highlighted_pdf_path, project_labels, project_titles, project_context_list

# --- MODIFIED: gradio_analyze_resume to handle project_context_list ---
def gradio_analyze_resume(resume_file, job_description, role):
    if resume_file is None or not job_description or not role:
        return "## Error\nPlease provide all inputs.", "<p style='color: red;'>Incomplete inputs.</p>", None, [], []
    if not llm_text:
         return "## Error\nLLM not initialized. Check GROQ_API_KEY.", "<p style='color: red;'>LLM Error.</p>", None, [], []

    temp_path = os.path.join("/tmp", os.path.basename(resume_file.name))
    shutil.copyfile(resume_file.name, temp_path)

    improvements, pdf_path, labels, project_titles, project_context_list = resume_analysis_pipeline(temp_path, job_description, role)

    eval_html = create_project_evaluation_html(labels)
    improvements_md = f"## Suggested Improvements\n\n{improvements}"

    # MODIFIED: Return project_context_list
    return improvements_md, eval_html, pdf_path, project_titles, project_context_list

# --- MODIFIED: generate_mock_questions for contextual questions and no-repeat logic ---
def generate_mock_questions(project_titles: List[str], project_context_list: List[Dict[str, str]], asked_questions: List[str], current_generate_btn_visibility):

    if not project_titles:
        updates = [gr.update(label="No projects available from resume analysis to generate questions.", choices=[], visible=True)]
        updates.extend([gr.update(visible=False) for _ in range(4)])
        return (*updates, gr.update(visible=False), asked_questions, {}, gr.update(visible=False))

    questions_data = []

    # Create a mapping from title to context for easier access
    context_map = {p['title']: p['description'] for p in project_context_list}

    # Use only the first 5 projects for the quiz set
    projects_to_quiz = project_titles[:5]

    for proj_title in projects_to_quiz:
        proj_context = context_map.get(proj_title, "No detailed description available.")

        task = Task(
            description=(
                f"Generate a new, unique technical multiple-choice question (MCQ) about the project: '{proj_title}'. "
                f"The question must focus on a key technical aspect, algorithm, or metric. "
                f"You MUST NOT ask any question that is semantically similar to the following previously asked questions: {asked_questions}. "
                f"Use the following project context for details: \n"
                f"--- PROJECT CONTEXT ---\n{proj_context}\n"
            ),
            agent=quiz_master_agent,
            expected_output="A valid JSON object matching the QuizQuestion schema: {'question': '...', 'options': ['...', '...', '...', '...'], 'correct_answer': '...'}",
        )
        crew = Crew(agents=[quiz_master_agent], tasks=[task], process=Process.sequential)

        try:
            raw_result = crew.kickoff()
            qa_data = safe_json_parse(str(raw_result), {})

            if qa_data and all(k in qa_data for k in ['question', 'options', 'correct_answer']):
                questions_data.append(qa_data)
                # APPEND the new question to the asked list
                asked_questions.append(qa_data['question'])
        except Exception as e:
            print(f"Error generating question for {proj_title}: {e}")
            continue

    # --- Process Results for Gradio Output ---
    questions = []
    options_list = []
    answers = []

    for qa_data in questions_data:
        questions.append(qa_data['question'])
        options_list.append(qa_data['options'])
        answers.append(qa_data['correct_answer'])

    state_data = {
        "questions": questions,
        "options": options_list,
        "answers": answers,
        "user_answers": [None] * len(questions)
    }

    updates = []
    num_questions = len(questions)

    if num_questions == 0:
        updates.append(gr.update(label="LLM failed to generate any valid questions.", choices=[], visible=True))
        updates.extend([gr.update(visible=False) for _ in range(4)])
        return (*updates, gr.update(visible=False), asked_questions, state_data, gr.update(visible=False))

    for i in range(5):
        if i < num_questions:
            current_options = state_data["options"][i]
            current_answer = state_data["answers"][i]
            if current_answer not in current_options:
                current_options.append(current_answer)
                random.shuffle(current_options)

            updates.append(gr.update(
                label=questions[i],
                choices=current_options,
                visible=True,
                value=None
            ))
        else:
            updates.append(gr.update(visible=False, value=None))

    # HIDE the initial "Generate First 5 Questions" button after the first successful run
    new_generate_btn_visibility = gr.update(visible=False)

    # MODIFIED RETURN: includes the updated asked_questions list
    return (*updates, gr.update(visible=True), asked_questions, state_data, new_generate_btn_visibility)

# --- Unchanged Mock Test Helpers ---
def evaluate_mock_answers(state_data: dict, *user_answers: str):
    state = state_data
    num_questions = len(state.get("questions", []))
    actual_answers = list(user_answers)[:num_questions]
    state["user_answers"] = actual_answers

    results_md = "<p style='color: white; font-size: 1.2em;'>**Mock Test Results**</p>"
    correct_count = 0

    for i, (ua, ca) in enumerate(zip(state["user_answers"], state["answers"])):
        ua_display = ua if ua is not None else "No Answer"
        is_correct = (ua == ca)
        color = "green" if is_correct else "red"
        if is_correct: correct_count += 1

        results_md += f"**Q{i+1}:** {state['questions'][i]}\n- Your answer: <span style='color:{color};'>{ua_display}</span>\n- Correct answer: {ca}\n\n"

    total_questions = len(state["questions"])
    score = f"**Score: {correct_count}/{total_questions}**"

    return (
        f"## Test Complete! {score}\n\n{results_md}",
        gr.update(visible=True),
        state
    )
